Creating a Job and CronJob     

Job 

apiVersion: batch/v1           <==== Batch API
kind: Job                      <==== Job Kind
metadata:
  name: pie-counter
spec:
  completions: 4  <== completions to be successful in order for the job as a whole to be successful, require 4 successful pod runs
  parallelism: 2  <== how many pods needs to run at the same time, 2 Pods can run in parallel at a time
  template:
    metadata:
      name: pie-counter
    spec:
      restartPolicy: Never      <==== Never try to restart(Never or OnFailure) -- if it fail it would try to restart the job OnFailure
      containers:
      - name: pie-counter
        image: alpine
        command:               <===== Job command to run   
        - "sh"
        - "-c"
        - "echo 'scale=1000; 4*a(1)' ...;"


# Create a new Job
k create -f file.job.yml --save-config
k apply -f pie-counter.job.yml 

# Creating or modifying a Job
k apply -f file.job.yml

apiVersion: batch/v1           <==== CronJob batch API
kind: CronJob                  <==== CronJob Kind
metadata:
  name: pie-counter
spec:
  concurrencyPolicy: Allow     <==== Allow multiple Pods to run even if their scheduling overlaps
  # Run the job every minute
  schedule "*/1 * * * *"       <==== Cron format to use for scheduling, its going to run every minute
  jobTemplate:
    spec:
      template:
        template
          spec:
            restartPolicy: OnFailure <=== Restart if there's a failure
            containers:
            - name: pie-counter
              image: alpine
              command:               <===== Command to run 
              - "sh"
              - "-c"
              - "echo 'scale=1000; 4*a(1)' ...;"



# Create a CronJob
A job can be created using the standard k create or k apply commands
k create -f file.cronjob.yml --save-config

# Creating or modifying a Job
k apply -f file.cronjob.yml
